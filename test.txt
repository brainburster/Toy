#定义数组
arr = [3,12,5,2,1,6,7,9,4,8,0,11,10]
#打印数组
echo("before sort:");
echo(arr)

#定义排序函数
func fast_sort(head, tail){
    if(head>=tail){
        return;
    }
    x = arr[head];
    i = head;
    j = tail;

    loop(i<j)
    {
        loop (i < j && arr[j] >= x){
            j=j-1;
        }
        if (i < j){
            i=i+1;
            arr[i] = arr[j];
        }

        loop (i < j && arr[i] < x){
            i=i+1;
        }
        if (i < j){
            j=j-1;
            arr[j] = arr[i];
        }
    }
    arr[i] = x;
    fast_sort(head, i-1);
    fast_sort(i+1, tail);
}

#排序
fast_sort(0,12);

#打印数组
echo("after sort:");
echo(arr)
